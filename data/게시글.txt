/*게시글 테이블 생성*/
create table posts(
    id number not null,
    title VARCHAR2(1000) not null,
    content VARCHAR2(2000),
    writer char(8),
    reg_date TIMESTAMP DEFAULT sysdate,
    primary key(id),
    foreign key(writer) references students(scode)
);

/*시퀀스생성*/
create sequence post_id start with 1 increment by 1;

/*최근 글순으로 행번호 출력 뷰생성*/
create view vposts as
select posts.*, sname, to_char(reg_date,'YYYY-MM-DD HH:mm:ss') fmt_date
from posts, students  
where writer=scode;

create view view_posts as
select rownum rn, p.* from
(select * from vposts order by id desc) p;

select * from view_posts where rn between 5 and 10;
